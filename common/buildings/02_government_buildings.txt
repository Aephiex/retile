# Bureaucratic Office
building_bureaucratic_1 = {
	base_buildtime = 360	
	base_cap_amount = 1
	can_be_disabled = no

	category = government
	
	potential = {
		exists = owner
		owner = { is_gestalt = no }
		has_district = district_bureaucrats
	}
	
	prerequisites = {
		"tech_adaptive_bureaucracy"
	}

	allow = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			AND = {
				retile_ai_planet_healthy_building = yes
				owner = { has_country_flag = retile_ai_needs_admin }
			}
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}
		upkeep = {
			energy = 2
		}
	}
	
	planet_modifier = {
		planet_stability_add = 1
	}
	
	upgrades = {
		building_bureaucratic_2
	}
	
	triggered_desc = {
		trigger = { exists = owner }
		text = retile_bureaucratic_1_effect
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	
	ai_weight = {
		weight = 100
	}
}

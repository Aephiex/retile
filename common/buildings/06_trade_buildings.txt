building_trade_1 = {
	base_buildtime = 360
	base_cap_amount = 1
	category = trade
	can_be_disabled = no
	
	potential = {
		exists = owner
		owner = { 
			is_regular_empire = yes 
			NOT = { has_tradition = tr_prosperity_public_works }
		}
		NOT = { has_modifier = slave_colony }
		OR = {
			has_district = district_clerks
			has_district = district_resort
		}
	}
	
	allow = {
		OR = {
			owner = { is_ai = no }
			AND = {
				retile_ai_planet_healthy_building = yes
				OR = {
					AND = {
						owner = { has_country_flag = retile_ai_needs_energy }
						count_pops = { limit = { has_job = clerk } count > 4 }
					}
					AND = {
						owner = { has_authority = auth_corporate }
						any_owned_pop = { has_job = clerk }
					}
				}
			}
		}
	}

	destroy_trigger = {
		OR = {
			owner = { is_regular_empire = no }
			has_modifier = slave_colony
			owner = { has_tradition = tr_prosperity_public_works }
		}
	}
	
	convert_to = {
		building_bourse_1
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}	
		upkeep = {
			energy = 1
		}		
	}
	
	prerequisites = {
		"tech_interplanetary_commerce"
	}
	
	show_tech_unlock_if = { 
		is_regular_empire = yes
		NOT = { has_tradition = tr_prosperity_public_works }
	}
	
	triggered_desc = {
		trigger = { always = yes }
		text = clerks_trade_1_effect
	}
	triggered_desc = {
		trigger = { always = yes }
		text = clerks_pollution_1_effect
	}
	
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_energy } }
		planet_clerks_energy_produces_add = @clerk_basic_1
	}
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_minerals } }
		planet_clerks_minerals_produces_add = @clerk_basic_1
	}
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_food } }
		planet_clerks_food_produces_add = @clerk_basic_1
	}
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_alloys } }
		planet_clerks_alloys_produces_add = @clerk_intermediate_1
	}
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_consumer_goods } }
		planet_clerks_consumer_goods_produces_add = @clerk_intermediate_1
	}
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_unity } }
		planet_clerks_unity_produces_add = @clerk_advanced_1
	}
	triggered_planet_modifier = {
		potential = { owner = { has_policy_flag = trade_conversion_research } }
		planet_clerks_physics_research_produces_add = @clerk_advanced_1
		planet_clerks_society_research_produces_add = @clerk_advanced_1
		planet_clerks_engineering_research_produces_add = @clerk_advanced_1
	}
	
	planet_modifier = {
		planet_clerks_energy_upkeep_add = @job_energy_normal
		planet_clerks_consumer_goods_upkeep_add = @job_supplies_normal
		clerks_bio_food_upkeep_add = @living_standard_food_normal
		clerks_lith_minerals_upkeep_add = @living_standard_food_normal
		clerks_mech_energy_upkeep_add = @living_standard_food_normal
	}
	
	country_modifier = {
		country_resource_max_add = 1000
	}
	
	upgrades = {
		"building_trade_2"
	}
	
	ai_weight = {
		weight = 100
	}
}

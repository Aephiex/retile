# This = pop
can_resettle_pop = {
	always = yes
	IF = { #Done as an IF/ELSE so the tooltip isn't hideous and unreadable
		limit = { is_shackled_robot = yes }
		is_shackled_robot = yes
	} ELSE = {
		owner = {
			OR = {
				has_policy_flag = resettlement_allowed
				is_gestalt = yes
			}
		}
	}
	planet = {
		num_pops > 1
	}
}

# This = planet
# Root = country
can_build_branch_office_on_planet = {
	custom_tooltip = {
		fail_text = requires_recipient_not_self
		exists = owner
		owner = {
			NOT = { is_same_value = root }
		}
	}
	custom_tooltip = {
		fail_text = requires_recipient_not_fanatic_purifiers
		exists = owner
		owner = {
			OR = {
				is_same_species = root
				NOT = { has_civic = civic_fanatic_purifiers }
			}
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_NOT_COMMERCIAL_PACT
		exists = owner
		OR = {
			root = { is_criminal_syndicate = yes }
			owner = { has_commercial_pact = root }
			owner = { is_in_federation_with = root }

			# allow branch offices in subsidiaries and vice versa
			root = {
				is_subject = yes
				#is_subject_type = subsidiary
				overlord = { is_same_value = prev.owner }
			}
			owner = {
				is_subject = yes
				#is_subject_type = subsidiary
				overlord = { is_same_value = root }
			}
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_CRIMINAL_HERITAGE_HAS_TRUCE
		exists = owner
		OR = {
			root = { is_criminal_syndicate = no }
			owner = {
				NOR = {
					has_truce = root
					is_at_war_with = root
				}
			}
		}
	}
	custom_tooltip = {
		fail_text = BRANCH_OFFICE_CRIMINAL_COOLDOWN
		exists = owner
		OR = {
			root = { is_criminal_syndicate = no }
			AND = {
				root = { is_criminal_syndicate = yes }
				NOT = { has_planet_flag = criminal_branch_closed_cooldown_planet@root }
			}
		}
	}
}

can_modify_species = {
	root = { OR = {	# prevent AI from spamming more than 3 subspecies...
		is_ai = no
		AND = {
			NOT = { has_country_flag = ai_recently_modified_species }
			count_exact_species = {
				count < 3
				limit = {
					OR = {
						is_same_species = root
						is_subspecies = root
					}
				}
			}
		}
	} }
	OR = {
		NOT = { has_trait = trait_self_modified }
		is_same_species = root
	}
}

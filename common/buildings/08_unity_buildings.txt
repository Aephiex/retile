building_autochthon_monument = {
	base_buildtime = 360
	base_cap_amount = 1
	category = unity
	can_be_disabled = no
	
	potential = {
		exists = owner
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = penal_colony }
		owner = { 
			is_gestalt = no
			is_megacorp = no
			NOR = {
				has_valid_civic = civic_exalted_priesthood
				has_valid_civic = civic_gospel_of_the_masses
				has_valid_civic = civic_imperial_cult
			}
		}			
		has_district = district_scholars
	}
	
	allow = {
		OR = {
			owner = { is_ai = no }
			AND = {
				retile_ai_planet_healthy_building = yes
				owner = { has_country_flag = retile_ai_needs_science }
				any_owned_pop = { has_job = scholar }
				OR = {
					has_building = building_research_lab_1
					has_building = building_research_lab_2
					has_building = building_research_lab_3
					has_building = building_metascience_lab_1
					has_building = building_metascience_lab_2
					has_building = building_metascience_lab_3
				}
			}
		}
	}

	destroy_trigger = {
		OR = {
			has_modifier = slave_colony
			has_modifier = penal_colony
			owner = { 
				OR = {
					is_gestalt = yes
					is_megacorp = yes
					has_valid_civic = civic_exalted_priesthood
					has_valid_civic = civic_gospel_of_the_masses
					has_valid_civic = civic_imperial_cult
				}
			}
			AND = {
				owner = { is_ai = yes }
				NOR = {
					has_building = building_research_lab_1
					has_building = building_research_lab_2
					has_building = building_research_lab_3
					has_building = building_metascience_lab_1
					has_building = building_metascience_lab_2
					has_building = building_metascience_lab_3
				}
			}
		}
	}
	
	convert_to = {
		building_temple
		building_hive_node
		building_uplink_node
		building_corporate_monument
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @cost_low
		}
		upkeep = {
			energy = 1
		}
	}
	
	prerequisites = {
		"tech_cultural_heritage"
	}	
	
	upgrades = {
		building_heritage_site
	}
	
	show_tech_unlock_if = {
		is_gestalt = no
		is_megacorp = no
		NOR = {
			has_valid_civic = civic_exalted_priesthood
			has_valid_civic = civic_gospel_of_the_masses
			has_valid_civic = civic_imperial_cult
		}
